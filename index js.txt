var sidenav; // Declare sidenav in the global scope

    function showNavbar() {
        sidenav = document.querySelector(".side-navbar");
        if (sidenav) {
            sidenav.style.left="0";
        } else {
            console.error("Element with class 'side-navbar' not found.");
        }
    }
    function closeNavbar() {
        sidenav = document.querySelector(".side-navbar");
        if (sidenav) {
            sidenav.style.left="-60%";
        } else {
            console.error("Element with class 'side-navbar' not found.");
        }
    }
    function showImage() {
       
        window.location.href = "download.jpg";
    }
